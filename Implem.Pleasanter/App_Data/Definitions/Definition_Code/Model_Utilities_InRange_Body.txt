private static bool InRange(Context context, SiteSettings ss, View view, int limit)
{
    var where = ss.DashboardParts?.Any() == true
        ? ss.DashboardParts[0].View.Where(context: context, ss: ss)
        : new SqlWhereCollection();
    where = view.Where(
        context: context,
        ss: ss,
        where: where);
    var param = view.Param(
        context: context,
        ss: ss);
    return Repository.ExecuteScalar_int(
        context: context,
        statements: Rds.Select#TableName#(
            column: Rds.#TableName#Column().#TableName#Count(),
            join: ss.Join(
                context: context,
                join: new IJoin[]
                {
                    where
                }),
            where: where,
            param: param)) <= limit;
}
