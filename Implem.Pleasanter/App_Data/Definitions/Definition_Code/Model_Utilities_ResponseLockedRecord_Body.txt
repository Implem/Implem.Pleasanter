else if (#modelName#Model.Locked)

    ss.SetLockedRecord(
        context: context,
        time: #modelName#Model.UpdatedTime,
        user: #modelName#Model.Updator);

        context: context,
        ss: ss,

                ss: ss,

                processes: processes))
            .Messages(context.Messages)
